Applied Code

This is the fix that was applied to the index.html:

// Ensure signup button works correctly after page loads
window.addEventListener('DOMContentLoaded', function() {
    setTimeout(function() {
        const signupBtn = document.getElementById('signupBtn');
        if (signupBtn) {
            // Remove any existing listeners
            const newBtn = signupBtn.cloneNode(true);
            signupBtn.parentNode.replaceChild(newBtn, signupBtn);
            
            // Add the correct listener
            newBtn.addEventListener('click', function(e) {
                e.preventDefault();
                e.stopPropagation();
                console.log('Signup button clicked - redirecting to signup page (fixed)');
                window.location.href = 'https://rasin.pyebwa.com/signup.html';
            });
            console.log('Signup button listener fixed');
        }
    }, 100);
});
