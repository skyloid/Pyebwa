        console.log('User document found by UID');
        console.log('Document ID:', user.uid);
        console.log('Current data:', doc.data());
        window.userDocId = user.uid;

        console.log('Checking by email...');
        const query = await db.collection('users').where('email', '==', user.email).get();
        if (!query.empty) {

            console.log('User document found by email');
            console.log('Document ID:', userDoc.id);
            console.log('Current data:', userDoc.data());
            window.userDocId = userDoc.id;
        } else {
            console.error('No user document found!');
        }
    }
})();
