# Firebase Rules Deployment Commands

# 1. First, ensure you have Firebase CLI installed:
npm install -g firebase-tools

# 2. Navigate to the project directory:
cd /home/pyebwa-rasin/htdocs/rasin.pyebwa.com

# 3. Login to Firebase (if not already logged in):
firebase login

# 4. Deploy the security rules:
firebase deploy --only firestore:rules,storage:rules --project pyebwa-f5960

# Alternative: Run the deployment script
./deploy-firebase-rules.sh

# To deploy indexes as well (recommended for performance):
firebase deploy --only firestore:rules,firestore:indexes,storage:rules --project pyebwa-f5960

# To test rules without deploying (dry run):
firebase deploy --only firestore:rules,storage:rules --project pyebwa-f5960 --dry-run
